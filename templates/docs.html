POST /interact

Record a user action on an item. Triggers recommendation generation automatically.

FieldTypeRequiredDescription
user_idstring Yes Your user's unique ID
item_idstring Yes Your product/item ID
actionstring Yes view, like, cart, or buy
regionstring No User's region (e.g. IN-TN)
categorystring No Item category
Python 
import requests

API_KEY = "{{ api_key }}"
BASE    = "{{ request.host_url }}api/v1"

def track_interaction(user_id, item_id, action,
                      region=None, category=None):
    try:
        requests.post(
            f"{BASE}/interact",
            headers={"X-API-Key": API_KEY,
                     "Content-Type": "application/json"},
            json={"user_id":  user_id,
                  "item_id":  item_id,
                  "action":   action,
                  "region":   region,
                  "category": category},
            timeout=2
        )
    except Exception:
        pass  # Never break your app if API is down

# Usage
track_interaction("user_123", "product_456", "buy",
                  region="IN-TN", category="electronics")

def track_interaction(user_id, item_id, action,

  uri  = URI("#{BASE}/interact")
  http = Net::HTTP.new(uri.host, uri.port)

  req.body = {

    category: category
  }.to_json
  http.request(req)
rescue
end

Java 
import java.net.http.*;
import java.net.URI;

public class RecAPI {
  static String API_KEY = "{{ api_key }}";
  static String BASE    = "{{ request.host_url }}api/v1";
  static HttpClient client = HttpClient.newHttpClient();

  public static void trackInteraction(
      String userId, String itemId, String action) {
    try {
      String body = String.format(
        "{\"user_id\":\"%s\",\"item_id\":\"%s\",\"action\":\"%s\"}",
        userId, itemId, action);
      HttpRequest req = HttpRequest.newBuilder()
        .uri(URI.create(BASE + "/interact"))
        .header("Content-Type", "application/json")
        .header("X-API-Key", API_KEY)
        .POST(HttpRequest.BodyPublishers.ofString(body))
        .build();
      client.send(req, HttpResponse.BodyHandlers.ofString());
    } catch (Exception e) {}
  }
}

GET /recommend

Fetch personalized recommendations for a user.

ParamType RequiredDescription
user_idstring Yes The user to recommend for
limitinteger No Max results (default 10)
Python 
def get_recommendations(user_id, limit=10):
    try:
        r = requests.get(
            f"{BASE}/recommend",
            headers={"X-API-Key": API_KEY},
            params={"user_id": user_id, "limit": limit},
            timeout=2
        )
        return r.json().get("recommendations", [])
    except Exception:
        return []

# Usage
recs = get_recommendations("user_123")
for item in recs:
    print(item["item_id"], item["score"])
JavaScript 
async function getRecommendations(userId, limit=10) {
  try {
    const res = await fetch(
      `${BASE}/recommend?user_id=${userId}&limit=${limit}`,
      { headers: { "X-API-Key": API_KEY } }
    );
    const data = await res.json();
    return data.recommendations || [];
  } catch (e) { return []; }
}

// Usage
const recs = await getRecommendations("user_123");
recs.forEach(r => console.log(r.item_id, r.score));

Action Weights

Each action carries a different weight in the recommendation algorithm.

ActionWeight Meaning
view 2 User viewed the item
like 4 User liked or saved
cart 8 User added to cart
buy 10 User completed purchase
